def area(a, b):
'''принимает значения a и b, а возвращает площадь прямоугольника со сторонами a и b'''
return a * b
def perimeter(a, b):
''' принимает значения a и b, а возвращает периметр прямоугольника со сторонами a и b''' 
return 2*(a + b)
